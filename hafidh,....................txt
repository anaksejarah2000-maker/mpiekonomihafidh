function BlokkKonten({ blok }) {
  switch (blok.tipe) {
    case "paragraf":
      return <p style={{ fontSize: 14, lineHeight: 1.85, color: C.text, margin: "0 0 12px" }}>{blok.teks}</p>;
    case "subjudul":
      return <p style={{ fontSize: 15, fontWeight: 700, color: C.primary, margin: "16px 0 6px" }}>{blok.teks}</p>;
    case "info-box":
      return (
        <div style={{ background: blok.warna, border: `1.5px solid ${blok.border}`, borderRadius: 10, padding: "12px 16px", margin: "10px 0 14px", fontSize: 13.5, lineHeight: 1.7, color: C.text }}>
          {blok.teks}
        </div>
      );
    case "list":
      return (
        <div style={{ margin: "10px 0 14px" }}>
          {blok.judul && <p style={{ fontWeight: 700, fontSize: 13.5, color: C.text, marginBottom: 6 }}>{blok.judul}</p>}
          <ul style={{ paddingLeft: 18, margin: 0 }}>
            {blok.item.map((it, i) => (
              <li key={i} style={{ fontSize: 13.5, lineHeight: 1.8, color: C.text, marginBottom: 3 }}>{it}</li>
            ))}
          </ul>
        </div>
      );
    case "diagram":
      return (
        <div style={{ display: "flex", alignItems: "center", justifyContent: "center", gap: 12, background: C.primaryLight, borderRadius: 12, padding: "16px 24px", margin: "12px 0", flexWrap: "wrap" }}>
          <div style={{ background: C.white, border: `2px solid ${C.primaryMid}`, borderRadius: 10, padding: "8px 18px", fontWeight: 700, fontSize: 13, color: C.primary }}>{blok.label1}</div>
          <span style={{ fontSize: 22, color: C.primaryMid }}>{blok.panah}</span>
          <div style={{ background: C.white, border: `2px solid ${C.primaryMid}`, borderRadius: 10, padding: "8px 18px", fontWeight: 700, fontSize: 13, color: C.primary }}>{blok.label2}</div>
        </div>
      );
    case "dua-kolom":
      return (
        <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 12, margin: "12px 0 16px" }}>
          {[blok.kiri, blok.kanan].map((sisi, i) => (
            <div key={i} style={{ background: sisi.warna, border: `1.5px solid ${sisi.border}`, borderRadius: 10, padding: "12px 14px" }}>
              <p style={{ fontWeight: 700, fontSize: 13.5, marginBottom: 6, color: C.text }}>{sisi.judul}</p>
              <ul style={{ paddingLeft: 16, margin: 0 }}>
                {sisi.item.map((it, j) => <li key={j} style={{ fontSize: 13, lineHeight: 1.75, color: C.text, marginBottom: 2 }}>{it}</li>)}
              </ul>
            </div>
          ))}
        </div>
      );
    case "tiga-kartu":
      return (
        <div style={{ display: "grid", gridTemplateColumns: "repeat(3,1fr)", gap: 12, margin: "12px 0 16px" }}>
          {blok.kartu.map((k, i) => (
            <div key={i} style={{ background: k.warna, border: `1.5px solid ${k.border}`, borderRadius: 10, padding: "12px 14px" }}>
              <p style={{ fontWeight: 700, fontSize: 13.5, marginBottom: 6, color: C.text }}>{k.judul}</p>
              {k.sub && <p style={{ fontWeight: 600, fontSize: 12, color: C.textMuted, marginBottom: 4 }}>{k.sub}</p>}
              <ul style={{ paddingLeft: 14, margin: 0 }}>
                {k.item.map((it, j) => <li key={j} style={{ fontSize: 12.5, lineHeight: 1.7, color: C.text, marginBottom: 1 }}>{it}</li>)}
              </ul>
              {k.sub2 && <p style={{ fontWeight: 600, fontSize: 12, color: C.textMuted, margin: "8px 0 4px" }}>{k.sub2}</p>}
              {k.item2 && <ul style={{ paddingLeft: 14, margin: 0 }}>{k.item2.map((it, j) => <li key={j} style={{ fontSize: 12.5, lineHeight: 1.7, color: C.text, marginBottom: 1 }}>{it}</li>)}</ul>}
            </div>
          ))}
        </div>
      );
    case "perbandingan":
      return (
        <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 12, margin: "12px 0 16px" }}>
          {[blok.kiri, blok.kanan].map((sisi, i) => (
            <div key={i} style={{ background: sisi.warna, border: `2px solid ${sisi.border}`, borderRadius: 12, padding: "14px 16px" }}>
              <p style={{ fontWeight: 800, fontSize: 14, color: sisi.border, marginBottom: 10, textAlign: "center" }}>{sisi.judul}</p>
              <ul style={{ paddingLeft: 16, margin: 0 }}>
                {sisi.item.map((it, j) => <li key={j} style={{ fontSize: 13, lineHeight: 1.8, color: C.text, marginBottom: 3 }}>{it}</li>)}
              </ul>
            </div>
          ))}
        </div>
      );
    default:
      return null;
  }
}
